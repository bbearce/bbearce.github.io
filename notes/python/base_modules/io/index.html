Source: w3schools 

import shutil

with open('from_here.txt', "w") as file:
    file.write("from here")

src="./from_here.txt"
dest="./to_here/"
shutil.move(src, dest)

Copy File

import shutil

src="from_here.txt"
dest="./to_here/"
shutil.copyfile(src, os.path.join(dest,src))
